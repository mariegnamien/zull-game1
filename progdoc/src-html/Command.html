003/**
004 * Classe Command - une commande du jeu d'aventure Zuul.
005 *
006 * @author Marie Emilienne Gnamoen
007 */
008public class Command
009{
010    private String aCommandWord;
011    private String aSecondWord;
012
013    /**
014     * Constructeur naturel qui définit une commande en fonction des données entrées par l'utilisateur.
015     * @param pWord1 premier mot de la commande
016     * @param pWord2 deuxieme mot de la commande
017     */
018    public Command(final String pWord1, final String pWord2){
019        this.aCommandWord = pWord1;
020        this.aSecondWord = pWord2;
021    }//Command(.)
022    /**
023     * @return Retourne le premier mot de la commande entrée.
024     */  
025    public String getCommandWord(){
026        return this.aCommandWord;
027    }//getCommandWord()
028    
029    /**
030        @return Retourne le second mot de la commande.
031     */
032    public String getSecondWord(){
033        return this.aSecondWord;
034    }//getSecondWord
035    
036    /**
037     * @return Retourne un booléen en fonction de si la commande possède un 2e mot ou non.
038     */
039    public boolean hasSecondWord(){
040        return(this.aSecondWord != null);
041    }//hasSecondWord()
042    
043    /**
044        @return Retourne un booléen en fonction de si la commande est reconnue ou non.
045     */
046    public boolean isUnknown(){
047        return(this.aCommandWord == null);        
048    }//isUnknown()
049} // Command
