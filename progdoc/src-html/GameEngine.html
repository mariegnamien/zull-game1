181        /**
182        * Permet de revenir dans la salle précédente une fois la commande entrée par l'utilisateur.
183        */
184    private void back(){
185        if(!this.aRoomStack.empty()){
186        this.aCurrentRoom = this.aRoomStack.pop();
187        this.printLocationInfo();
188        }
189        else{
190            this.aGui.println("You can't go back...");
191        return;
192        }
193    }//back()
194    
195        /**
196        *
197        * Retourne un booléen après traitement de la commande entrée par l'utilisateur.
198        * @param pWord Commande entrée par l'utilisateur.
199        */
200       
201        public void interpretCommand(final String pWord){
202        this.aGui.println( "> " + pWord );
203        Command vCommand = this.aParser.getCommand(pWord); 
204        
205        if(vCommand.isUnknown()) {
206            this.aGui.println("I don't know what you mean ?");
207            return;
208        }
209        
210        String vCommandWord = vCommand.getCommandWord();
211    
212        if(vCommandWord.equals("help")){
213            this.printHelp();
214        }
215
216        else if(vCommandWord.equals("go")){
217            this.aRoomStack.push(this.aCurrentRoom);
218            this.goRoom(vCommand);
219        }
220        else if(vCommandWord.equals("look")){
221            this.look();
222        }
223        else if(vCommandWord.equals("eat")){
224            if(vCommand.hasSecondWord()){
225                this.aGui.println("You can't do that");
226            }
227            else{
228            this.eat();
229        }
230        }
231        
232        else if(vCommandWord.equals("back")){
233            if(vCommand.hasSecondWord()){
234                this.aGui.println("You can't do that...");
235            }
236            else{
237                this.back();
238            }
239        }
240        
241        else if(vCommandWord.equals("start!")){
242            this.printLocationInfo();
243        }
244        
245        else if (vCommandWord.equals("quit")){
246            if (vCommand.hasSecondWord()){
247                this.aGui.println( "Quit what?" );
248            }
249            else{
250                this.endGame();
251            }
252        }
253        else{
254            this.aGui.println("Erreur du programmeur : commande non reconnue !");
255        }
256    }//interpretCommand()

270    /**
271    *
272    * Permet de se déplacer entre les différentes pièces du jeu.
273    * @param pRoom pièce dans laquelle l'utilisateur veut se rendre
274    */
275   
276    private void goRoom(final Command pRoom){
277        Room vNextRoom = null;
278        String vDirection = pRoom.getSecondWord();
279        
280        if(!pRoom.hasSecondWord()){
281                this.aGui.println("Go where ?");
282                return;
283            }
284        vNextRoom = aCurrentRoom.getExit(vDirection);
285        
286        if(vNextRoom != null){
287            this.aCurrentRoom = vNextRoom;
288            this.aGui.println(this.aCurrentRoom.getLongDescription());
289            if(this.aCurrentRoom.getImageName() != null){
290                this.aGui.showImage(this.aCurrentRoom.getImageName());
291            }
292        }
293        else if(vNextRoom == null){
294            this.aGui.println("There is no path!");
295            return;
296        }
297        else{
298            this.aGui.println("Unknown direction !");
299        }
300    } // Game
