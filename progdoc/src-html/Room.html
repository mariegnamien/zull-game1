001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * Classe Room - un lieu du jeu d'aventure Zuul.
006 *
007 * @author Gnamien Marie Emilienne 
008 */
009public class Room
010{
011    private String aDescription;
012    private HashMap<String, Room> aExits;
013    private String aImageName;
014    private HashMap<String,Item> aItems;
015    
016    /**
017    *
018    * Crée une pièce avec une description particulière.
019    * @param pDescription description de la pièce
020    * @param pImage nom d'une image
021    */
022
023    public Room(final String pDescription, final String pImage){
024        this.aDescription = pDescription;
025        this.aExits = new HashMap<String,Room>();
026        this.aItems = new HashMap<String,Item>();
027        this.aImageName = pImage;
028    }// Room()
029    
030    /**
031    *
032    * Retourne la description d'une pièce.
033    * @return Retourne la description de la pièce courante.
034    * 
035    */
036    public String getDescription(){
037        return this.aDescription;
038    }//getDescription()
039
040    /**
041    *
042    * Définit une des sorties de la pièce.
043    * @param pDescription nom de la sortie
044    * @param pExit pièce en sortie de la pièce courante
045    */
046    public void setExit(final String pDescription, final Room pExit){
047        this.aExits.put(pDescription,pExit);
048    }//setExit(.)
049
050    /**
051    *
052    *  @return Retourne la sortie d'une pièce en fonction de la direction entrée.
053    * @param pDirection pièce en sortie
054    */
055    public Room getExit(String pDirection){
056        return this.aExits.get(pDirection);
057    }//getExit(.)
058   
059    /**
060    *
061    * @return Retourne l'ensemble des sorties d'une pièce.
062    * 
063    *
064    */
065   
066    public String getExitString(){
067        StringBuilder returnString = new StringBuilder( "Exits :" );
068        for(String vKeys : this.aExits.keySet()){
069            if(aExits.get(vKeys) != null){
070            returnString.append( " " + vKeys );
071        }
072        }
073        return returnString.toString();
074    }//getExitString()
075    
076    /** Procédure qui ajoute un Item à l'HashMap aItems
077     * @param pNom nom de l'item
078     * @param pItem Item
079    */
080    public void addItem(final String pNom, final Item pItem){
081        this.aItems.put(pNom,pItem);
082    }//addItem(.)
083    
084    /** @return Retourne la liste de tous les Items présents dans la pièce courante
085    */
086    public String getItemString(){
087        StringBuilder vReturn = new StringBuilder("Items : ");
088        if(this.aItems.keySet().size() == 0){
089            return "No Items here...";
090        }
091        
092        for(String nom : this.aItems.keySet()){
093            vReturn.append(this.aItems.get(nom).getItemDescription() + "; ");
094        }
095        return vReturn.toString();
096        }//getItemString()
097    
098    /**
099     * Retourne une longue description de cette "room", de la forme 
100     *          You are in the kitchen
101     *          Exits : north west.
102     * @return A description of the room, including exits. 
103     */
104    public String getLongDescription(){
105        return "You are " + this.aDescription + ".\n" + this.getExitString() + "\n" + this.getItemString();
106    }//getLongDescription()
107    
108    /** @return Retourne le nom d'une Image.
109    */
110    public String getImageName(){
111        return this.aImageName;
112    }//getImageName()
113
114} // Room
