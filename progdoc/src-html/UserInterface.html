001import javax.swing.JFrame;
002import javax.swing.JTextField;
003import javax.swing.JTextArea;
004import javax.swing.JLabel;
005import javax.swing.ImageIcon;
006import javax.swing.JScrollPane;
007import javax.swing.JPanel;
008import javax.swing.JButton;
009
010import java.awt.Dimension;
011import java.awt.BorderLayout;
012
013import java.awt.event.ActionListener;
014import java.awt.event.WindowAdapter;
015import java.awt.event.WindowEvent;
016import java.awt.event.ActionEvent;
017
018import java.net.URL;
019
020
021/**
022 * This class implements a simple graphical user interface with a 
023 * text entry area, a text output area and an optional image.
024 * 
025 * @author Michael Kolling
026 * @version 1.0 (Jan 2003) DB edited (2023)
027 */
028public class UserInterface implements ActionListener
029{
030    private GameEngine aEngine;
031    private JFrame     aMyFrame;
032    private JTextField aEntryField;
033    private JTextArea  aLog;
034    private JLabel     aImage;
035    private JButton aButton;
036    
037    /**
038     * Construct a UserInterface. As a parameter, a Game Engine
039     * (an object processing and executing the game commands) is
040     * needed.
041     * 
042     * @param pGameEngine The GameEngine object implementing the game logic.
043     */
044    public UserInterface( final GameEngine pGameEngine )
045    {
046        this.aEngine = pGameEngine;
047        this.createGUI();
048    } // UserInterface(.)
049
050    /**
051     * Print out some text into the text area.
052     * @param pText chaine de caracteres
053     */
054    public void print( final String pText )
055    {
056        this.aLog.append( pText );
057        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
058    } // print(.)
059
060    /**
061     * Print out some text into the text area, followed by a line break.
062     * @param pText texte écrit dans la fenêtre
063    */
064    public void println( final String pText )
065    {
066        this.print( pText + "\n" );
067    } // println(.)
068
069    /**
070     * Show an image file in the interface.
071     * @param pImageName nom de l'image
072     */
073    public void showImage( final String pImageName )
074    {
075        String vImagePath = "" + pImageName; // to change the directory
076        URL vImageURL = this.getClass().getClassLoader().getResource( vImagePath );
077        if ( vImageURL == null )
078            System.out.println( "Image not found : " + vImagePath );
079        else {
080            ImageIcon vIcon = new ImageIcon( vImageURL );
081            this.aImage.setIcon( vIcon );
082            this.aMyFrame.pack();
083        }
084    } // showImage(.)
085
086    /**
087     * Enable or disable input in the entry field.
088     * @param pOnOff booleen
089     */
090    public void enable( final boolean pOnOff )
091    {
092        this.aEntryField.setEditable( pOnOff ); // enable/disable
093        if ( pOnOff ) { // enable
094            this.aEntryField.getCaret().setBlinkRate( 500 ); // cursor blink
095            this.aEntryField.addActionListener( this ); // reacts to entry
096        }
097        else { // disable
098            this.aEntryField.getCaret().setBlinkRate( 0 ); // cursor won't blink
099            this.aEntryField.removeActionListener( this ); // won't react to entry
100        }
101    } // enable(.)
102
103    /**
104     * Set up graphical user interface.
105     */
106    private void createGUI()
107    {
108        this.aMyFrame = new JFrame( "3417." );
109        this.aEntryField = new JTextField( 34 );
110        this.aEntryField.enable(false);
111        
112        this.aLog = new JTextArea();
113        this.aLog.setEditable( false );
114        JScrollPane vListScroller = new JScrollPane( this.aLog );
115        vListScroller.setPreferredSize( new Dimension(200, 200) );
116        vListScroller.setMinimumSize( new Dimension(100,100) );
117        
118        this.aButton = new JButton("start!");// bouton de départ
119        
120        aMyFrame.add(this.aButton,BorderLayout.EAST); // positionné à droite de la fenêtre
121
122        this.aImage = new JLabel();
123
124        JPanel vPanel = new JPanel();
125        vPanel.setLayout( new BorderLayout() ); // ==> only five places
126        vPanel.add( this.aImage, BorderLayout.NORTH );
127        vPanel.add( vListScroller, BorderLayout.CENTER );
128        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
129
130        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.CENTER );
131
132        // add some event listeners to some components
133        this.aEntryField.addActionListener( this );
134        this.aButton.addActionListener(this);
135        
136        // to end program when window is closed
137        this.aMyFrame.addWindowListener(
138            new WindowAdapter() { // anonymous class
139                /** Ferme la fenêtre du jeu.
140                 * @param pE WindowEvent
141                 */
142                @Override public void windowClosing(final WindowEvent pE)
143                {
144                    System.exit(0);
145                }
146        } );
147
148        this.aMyFrame.pack();
149        this.aMyFrame.setVisible( true );
150        this.aEntryField.requestFocus();
151    } // createGUI()
152
153    /**
154     * Actionlistener interface for entry textfield.
155     * @param pE ActionEvent
156     */
157    @Override public void actionPerformed( final ActionEvent pE ) 
158    {
159        if(pE.getSource() == this.aButton){ // si le bouton est utilisé
160            aMyFrame.remove(aButton); // le bouton est retiré
161            this.aEntryField.setText( "start!" ); // et le texte start considéré comme commande est automatiquement entrée dans l'Entry Field
162            this.aEntryField.enable(true); // l'EntryField est ensuite activée pour être utilisable par l'utilisateur
163        }
164        // because there is only one possible action (text input) :
165        this.processCommand(); // never suppress this line
166    } // actionPerformed(.)
167
168    /**
169     * A command has been entered in the entry field.  
170     * Read the command and do whatever is necessary to process it.
171     */
172    private void processCommand()
173    {
174        String vInput = this.aEntryField.getText();
175        this.aEntryField.setText( "" );
176        
177        this.aEngine.interpretCommand( vInput );
178    } // processCommand()
179} // UserInterface
